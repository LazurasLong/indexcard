\documentclass[4pt,twocolumn,landscape]{article}
% Packages should be included with repo, but you can find
% them all at http://ctan.org/
\usepackage{multicol,fontspec,geometry,titlesec,enumitem}

% remove margins/spacing from lists
\setlist{noitemsep,nolistsep,leftmargin=1em,labelsep=0pt,topsep=0pt}

% remove paragraph indent, add one line of spacing
% between paragraphs
\setlength{\parindent}{0pt}
\setlength{\parskip}{1em}

% set the margin between the two columns on a card
\columnsep=2em

% defines tightlist; this is a bugfix for pandoc
\def\tightlist{}

% new page every section
\let\stdsection\section  % avoids recursion error
\renewcommand\section{\clearpage\stdsection}

%two column headers
\let\newsection\section  % avoids recursion error
\renewcommand\section{\clearpage\newsection}

% fonts
\setmainfont[BoldFont={Merriweather-Bold.ttf},
             ItalicFont={Merriweather-Italic.ttf},
             BoldItalicFont={Merriweather-BoldItalic.ttf},
             SmallCapsFont={Florsn30.ttf}]
            {Merriweather-Regular.ttf}
\setsansfont{MerriweatherSans-Bold.ttf}
\setmonofont{AnticSlab-Regular.ttf}

% format headers
\titleformat{\section}
  {\normalfont\cleardoublepage\Large\sffamily\filcenter\bfseries}
  {}{0pt}{}
\titlespacing*{\section}{0pt}{0pt}{0pt}

% page dimensions, margins
\geometry{paperwidth=5in,paperheight=3in,tmargin=0.2in,
  bmargin=0.2in,lmargin=0.2in,rmargin=0.2in}

% the document itself
\begin{document}
$body$
\end{document}
